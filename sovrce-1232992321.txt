#!/usr/bin/perl

@A = (0, 6);
@B = (0, 1);
@C = ("", "(0,1)");

$N = 100000;
for ($i = 2; $i < $N; $i++) {
	$A[$i] = 5 + $i;
	$B[$i] = 1;
	$C[$i] = "(0,$i)";
	for ($j = 3; $j <= $i / 2; $j++) {
		$k = int($i / $j);
		$l = $i - $j * $k;
		$tmp = $A[$j] + 5 + 2 * $k + $l;
		if ($A[$i] > $tmp) {
			$A[$i] = $tmp;
			$B[$i] = $B[$j] + 1;
			$C[$i] = $C[$j] . " ($k,$l)";
		}
	}
}

for ($n = 1; $n < $N; $n++) {
	printf("% 5d: % 5d %d %s\n", $n, $A[$n], $B[$n], $C[$n]);
}
